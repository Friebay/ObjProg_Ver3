76// Skaičiuoja pažymių medianą
77float skaiciuotiMediana(const std::vector<int> &pazymiai)
78{
79 if (pazymiai.empty())
80 return 0;
81
82 std::vector<int> tempPazymiai = pazymiai; // Create a copy to avoid modifying original
83 size_t n = tempPazymiai.size();
84 size_t middle = n / 2;
85
86 if (n % 2 == 0)
87 {
88 std::nth_element(tempPazymiai.begin(), tempPazymiai.begin() + middle - 1, tempPazymiai.end());
89 std::nth_element(tempPazymiai.begin() + middle - 1, tempPazymiai.begin() + middle, tempPazymiai.end());
90 return (tempPazymiai[middle - 1] + tempPazymiai[middle]) / 2.0f;
91 }
92 else
93 {
94 std::nth_element(tempPazymiai.begin(), tempPazymiai.begin() + middle, tempPazymiai.end());
95 return tempPazymiai[middle];
96 }
97}
